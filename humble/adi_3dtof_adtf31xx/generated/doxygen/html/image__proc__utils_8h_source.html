55 void generateRangeTo3DLUT(CameraIntrinsics * camera_intrinsics)
56 {
57 /* Generate Camera Intrinsics calibration arrays. */
58 double k_raw_array[9] = {
59 camera_intrinsics->camera_matrix[0], camera_intrinsics->camera_matrix[1],
60 camera_intrinsics->camera_matrix[2], camera_intrinsics->camera_matrix[3],
61 camera_intrinsics->camera_matrix[4], camera_intrinsics->camera_matrix[5],
62 camera_intrinsics->camera_matrix[6], camera_intrinsics->camera_matrix[7],
63 camera_intrinsics->camera_matrix[8]};
64 double d_raw_array[8] = {
65 camera_intrinsics->distortion_coeffs[0], camera_intrinsics->distortion_coeffs[1],
66 camera_intrinsics->distortion_coeffs[2], camera_intrinsics->distortion_coeffs[3],
67 camera_intrinsics->distortion_coeffs[4], camera_intrinsics->distortion_coeffs[5],
68 camera_intrinsics->distortion_coeffs[6], camera_intrinsics->distortion_coeffs[7]};
69
70 cv::Mat k_raw = cv::Mat(3, 3, CV_64F, k_raw_array);
71 cv::Mat d_raw = cv::Mat(1, 8, CV_64F, d_raw_array);
72 cv::Size img_size(image_width_, image_height_);
73 cv::Mat k_rect = cv::getOptimalNewCameraMatrix(k_raw, d_raw, img_size, 0, img_size, nullptr);
74
75 // Prepare the rectification maps
76 cv::Mat r = cv::Mat::eye(3, 3, CV_32F);
77
78 float * range_to_3d_lut = range_to_3d_lut_;
79
80 for (int y = 0; y < image_height_; y++) {
81 for (int x = 0; x < image_width_; x++) {
82 cv::Mat distorted_pt(1, 1, CV_32FC2, cv::Scalar(x, y));
83 cv::Mat undistorted_pt(1, 1, CV_32FC2);
84
85 cv::undistortPoints(distorted_pt, undistorted_pt, k_raw, d_raw);
86
87 float ux = undistorted_pt.at<float>(0);
88 float uy = undistorted_pt.at<float>(1);
89
90 float scale_factor = 1.0f / sqrtf(1.0f + ux * ux + uy * uy);
91
92 *range_to_3d_lut++ = (ux * scale_factor);
93 *range_to_3d_lut++ = (uy * scale_factor);
94 *range_to_3d_lut++ = (scale_factor);
95 }
96 }
97 }

106 void computePointCloud(unsigned short * range_image, short * xyz_frame)
107 {

110 for (int i = 0; i < image_height_; i++) {

113 *xyz_frame++ = (*range_to_3d_lut++) * (*range_image);

116 }
