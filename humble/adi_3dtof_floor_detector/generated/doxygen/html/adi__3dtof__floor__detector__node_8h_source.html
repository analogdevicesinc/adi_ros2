230 // Get input sensor module
231 input_sensor_ = InputSensorFactory::getInputSensor(input_sensor_mode_);
232
233 // Open the sensor
234 if (input_sensor_mode_ != 3) {
235 // If the mode is not ADTF31xx sensor over Network, then the ip should be set to ""
236 input_sensor_ip_.clear();
237 }
238 input_sensor_->openSensor(
239 input_file_name_, input_image_width, input_image_height, config_file_name_of_tof_sdk, input_sensor_ip_);
240 if (!input_sensor_->isOpened()) {
241 RCLCPP_ERROR(
242 this->get_logger(), "Could not open the sensor %s",
243 input_file_name_.c_str());
244 rclcpp::shutdown();
245 }
246
247 // Configure the sensor
248 input_sensor_->configureSensor(camera_mode);
249
250 image_width_ = input_sensor_->getFrameWidth();
251 image_height_ = input_sensor_->getFrameHeight();
252
253 // Buffer allocation
254 floor_mask_ = new unsigned char[image_width_ * image_height_];
255 compressed_floor_pixels_ = (image_width_ * image_height_) / 8;
256 if (enable_compression_op_image_topics_) {
257 compressed_floor_mask_ = new unsigned char[compressed_floor_pixels_];
258 }
259
260 // Create TF listerner instance
261 camera_map_tf_buffer_ = std::make_unique<tf2_ros::Buffer>(this->get_clock());
262 camera_map_tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*camera_map_tf_buffer_);
263
264 optical_map_tf_buffer_ = std::make_unique<tf2_ros::Buffer>(this->get_clock());
265 optical_map_tf_listener_ =
266 std::make_shared<tf2_ros::TransformListener>(*optical_map_tf_buffer_);
267
268 // Get output sensor module
269 output_sensor_ = OutputSensorFactory::getOutputSensor(output_sensor_mode_);
270 if (output_sensor_ != nullptr) {
271 output_sensor_->open(
272 input_file_name_, image_width_, image_height_, EnableAllOutputs);
273 }
274
275 // Get intrinsics and extrinsics
276 input_sensor_->getIntrinsics(&depth_intrinsics_);
277 input_sensor_->getExtrinsics(&depth_extrinsics_);
278
279 // Get camera link TFs
280 getCameraLinksTF();
281
282 // Object creation
283 image_proc_utils_ = new ImageProcUtils(&depth_intrinsics_, image_width_, image_height_);
284
285 floor_plane_detection_ = new FloorPlaneDetection(
286 image_width_, image_height_, ransac_distance_threshold_mtr_, ransac_max_iterations_,
287 discard_distance_threshold_mtr_, camera_height_mtr_);
288
289 // Publishers
290 if (enable_compression_op_image_topics_ == true) {
291 compressed_depth_image_publisher_ = this->create_publisher<sensor_msgs::msg::CompressedImage>(
292 "depth_image/compressedDepth", 10);
293 compressed_ab_image_publisher_ =
294 this->create_publisher<sensor_msgs::msg::CompressedImage>("ab_image/compressedDepth", 10);
295 compressed_floor_mask_publisher_ =
296 this->create_publisher<sensor_msgs::msg::CompressedImage>("compressed_floor_mask", 10);
297 } else {
298 depth_image_publisher_ = this->create_publisher<sensor_msgs::msg::Image>("depth_image", 10);
299 ab_image_publisher_ = this->create_publisher<sensor_msgs::msg::Image>("ab_image", 10);
300 floor_mask_publisher_ = this->create_publisher<sensor_msgs::msg::Image>("floor_mask", 10);
301 }
302 depth_info_publisher_ = this->create_publisher<sensor_msgs::msg::CameraInfo>("camera_info", 10);
303 if (enable_pointcloud_publisher_) {
304 xyz_image_publisher_ =
305 this->create_publisher<sensor_msgs::msg::PointCloud2>("point_cloud", 10);
306 }
307
308 // For File-io, we do not want to miss any frame, so increasing the queue size.
309 if (input_sensor_mode_ != 0) {
310 max_input_queue_length_ = 100;
311 max_output_queue_length_ = 100;
312 }
313
314 // setting ab thresold to pulsatrix
315 input_sensor_->setABinvalidationThreshold(ab_threshold_);
316
317 // setting confidence threshold to pulsatrix
318 input_sensor_->setConfidenceThreshold(confidence_threshold_);
319 }

636 rcl_interfaces::msg::SetParametersResult parametersCallback(
637 const std::vector<rclcpp::Parameter> & parameters)
638 {
639 rcl_interfaces::msg::SetParametersResult result;
640 result.successful = true;
641 result.reason = "success";
642 // Copy the parameters vector to a local variable.
643 for (const auto & param : parameters) {
644 if (param.get_name() == "param_enable_ransac_floor_detection") {
645 if (dynamic_reconf_params_.enable_ransac_floor_detection != param.as_bool()) {
646 dynamic_reconf_params_.enable_ransac_floor_detection = param.as_bool();
647 RCLCPP_INFO(
648 this->get_logger(), "The value of param_enable_ransac_floor_detection is changed to %s",
649 param.value_to_string().c_str());
650 }
651 }
652
653 else if (param.get_name() == "param_ransac_max_iterations") {
654 if (dynamic_reconf_params_.ransac_max_iterations != param.as_int()) {
655 dynamic_reconf_params_.ransac_max_iterations = param.as_int();
656 RCLCPP_INFO(
657 this->get_logger(), "The value of param_ransac_max_iterations is changed to %s",
658 param.value_to_string().c_str());
659 }
660 }
661
662 else if (param.get_name() == "param_ransac_distance_threshold_mtr") {
663 if (dynamic_reconf_params_.ransac_distance_threshold_mtr != (float)param.as_double()) {
664 dynamic_reconf_params_.ransac_distance_threshold_mtr = (float)param.as_double();
665 RCLCPP_INFO(
666 this->get_logger(), "The value of param_ransac_distance_threshold_mtr is changed to %s",
667 param.value_to_string().c_str());
668 }
669 }
670
671 else if (param.get_name() == "param_enable_fallback_floor_detection") {
